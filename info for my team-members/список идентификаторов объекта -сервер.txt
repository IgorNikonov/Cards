Сноска:  по свойству HTMLelement.Name происходит идентификация нужных полей формы, при считывании из них значений.

doctorSelect        id: "select-doctor"    HTMLelement формы: "selectDoctor"

visitorPatrName     id: "visitor-patr-name" elementName: "visitorPatrName"

visitPurpose        id: "purpose"            elementName: "purpose"
visitDescription    id: "description"       elementName: "description"
urgency             id: "urgency"           elementName: "urgency"
